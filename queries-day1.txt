CREATE TABLE product_cat(
id int NOT NULL AUTO_INCREMENT,
product_cat varchar(255) NOT NULL,
short_code varchar(20) NOT NULL,
PRIMARY KEY(id)
);

create table products( 
id int not null AUTO_INCREMENT, 
product_cat_id int not null, 
product varchar(255) not null, 
short_code varchar(20) not null, 
PRIMARY KEY(id), 
FOREIGN KEY(product_cat_id) REFERENCES product_cat(id) 
);

=== select product categories not in the products ===

select product_cat from product_cat where id not in (select distinct product_cat_id from products);


=== add ratings column ====

alter table products add ratings decimal(3,2);

update products set ratings=5 where id=2;

update products set ratings=4 where id=3;
update products set ratings=1 where id=1;

==== display product name whose product rating is above average ===

select product from products where ratings >= (select AVG(ratings) from products);

==== views ====

=== ratings >= 4 -> good 
or rating <4 and ratings>=3 -> Satisfactory 
or ratings <=3 -> Bad  =====


select product, ratings, CASE when ratings >= 4 then "Good" when ratings <4 and ratings>=3 then "Satisfactory" when ratings <=3 then "Bad" END AS COMMENT from products;
